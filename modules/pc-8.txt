import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      debugShowCheckedModeBanner: false,
      home: SlideImageScreen(),
    );
  }
}

class SlideImageScreen extends StatefulWidget {
  const SlideImageScreen({super.key});

  @override
  State<SlideImageScreen> createState() => _SlideImageScreenState();
}

class _SlideImageScreenState extends State<SlideImageScreen>
    with SingleTickerProviderStateMixin {
  late AnimationController _controller;
  late Animation<Offset> _animation;

  @override
  void initState() {
    super.initState();

    _controller = AnimationController(
      vsync: this,
      duration: const Duration(seconds: 1),
    );

    _animation = Tween<Offset>(
      begin: const Offset(-1.0, 0.0),
      end: Offset.zero,               
    ).animate(CurvedAnimation(
      parent: _controller,
      curve: Curves.easeInOut,
    ));

    _controller.forward(); 
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Slide Image Animation')),
      body: Center(
        child: SlideTransition(
          position: _animation,
          child: SizedBox(
            width: 250,
            height: 250,
            child: Image.network(
              'https://via.placeholder.com/250', // replace with your image link / AssetImage
              fit: BoxFit.cover,
            ),
          ),
        ),
      ),
    );
  }
}
